function doGet(e) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var logSheet = ss.getSheetByName("Sheet1");

  var now = new Date();
  var type  = e.parameter.type || "Unknown";
  var delta = Number(e.parameter.delta) || 0;
  var total = Number(e.parameter.total) || 0;
  var fill  = Number(e.parameter.fill) || 0;

  logSheet.appendRow([now, type, delta, total]);

  updateSummary(type, delta, total, fill);

  return ContentService.createTextOutput("OK");
}

function updateSummary(type, delta, total, fill) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var sheet = ss.getSheetByName("Sheet2"); // ⚠️ 

  var today = Utilities.formatDate(new Date(), "GMT+7", "yyyy-MM-dd");
  var data = sheet.getDataRange().getValues();

  var found = false;

  for (var i = 1; i < data.length; i++) {
    if (data[i][0] == today) {
      found = true;

      var plastic = Number(data[i][1]) || 0;
      var metal   = Number(data[i][2]) || 0;

      if (type == "Plastic") plastic += delta;
      if (type == "Metal") metal += delta;

      sheet.getRange(i + 1, 2).setValue(plastic);
      sheet.getRange(i + 1, 3).setValue(metal);
      sheet.getRange(i + 1, 4).setValue(Number(total));
      sheet.getRange(i + 1, 5).setValue(Number(fill));

      break;
    }
  }

  if (!found) {
    var plastic = (type == "Plastic") ? delta : 0;
    var metal   = (type == "Metal") ? delta : 0;

    sheet.appendRow([
      today,
      plastic,
      metal,
      total,
      fill
    ]);
  }
}
